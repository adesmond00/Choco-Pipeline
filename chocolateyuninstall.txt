$ErrorActionPreference = 'Stop'

$packageArgs = @{
    packageName   = $env:ChocolateyPackageName
    softwareName  = 'Google Chrome*'
    fileType      = 'MSI'
    silentArgs    = "/qn /norestart"
    validExitCodes= @(0, 3010, 1605, 1614, 1641)
}

# Find Chrome in the registry
[array]$key = Get-UninstallRegistryKey -SoftwareName $packageArgs['softwareName']

if ($key.Count -eq 1) {
    $key | % {
        $packageArgs['file'] = "$($_.UninstallString)"
        
        if ($packageArgs['fileType'] -eq 'MSI') {
            # For MSI: use the product code GUID
            $packageArgs['silentArgs'] = "$($_.PSChildName) $($packageArgs['silentArgs'])"
            $packageArgs['file'] = ''
        } else {
            # For EXE: might need to handle the uninstall string differently
            # Chrome's uninstaller is typically an EXE even if installed via MSI
            if ($packageArgs['file'] -match '\.exe') {
                $packageArgs['fileType'] = 'EXE'
                $packageArgs['silentArgs'] = '--uninstall --force-uninstall'
            }
        }
        
        Uninstall-ChocolateyPackage @packageArgs
    }
} elseif ($key.Count -eq 0) {
    Write-Warning "Google Chrome has already been uninstalled by other means."
} elseif ($key.Count -gt 1) {
    Write-Warning "$($key.Count) matches found!"
    Write-Warning "To prevent accidental data loss, no programs will be uninstalled."
    Write-Warning "Please alert package maintainer the following keys were matched:"
    $key | % {Write-Warning "- $($_.DisplayName)"}
}

Write-Host "Google Chrome Enterprise uninstallation completed" -ForegroundColor Green
